jQuery.validate使用方法

自带验证规则：

  1. 'zh-cn' 中文验证

  2. "phone" 验证手机号码

  3. "string" 字母

  4. "number" 验证纯数字

  5. "email" 验证电子邮箱

使用方法：

基本输入框

HTML代码

<div class="form-group common">
<label>中文验证：</label>
<input type="text" name='a' class='form-control' >
<div class="tip"></div>
</div>
<div class="form-group common">
<label>中文纯数字：</label>
<input type="text" name='b' class='form-control' >
<div class="tip"></div>
</div>

JS代码：

validate({ 调用验证
id:"na", 验证器名称,方便以后调用
tipPlacement:function(element,tip){ 提示信息放置函数
element.closest('.common').find('.tip').html(tip);将提示信息放在哪里，element为input节点，tip为提示内容
},
rules:{
'a':{ name为a的需要验证的节点
wrong:"请输入中文", wrong,right,empty,focus这几个字段为提示内容
right:"输入正确",
empty:'输入不能为空',
focus:"请您输入该字段",
required:true, 是否为必填(填了就得写对！)
limit:/^[\u4e00-\u9fa5]+$/ 验证正则
}
'b':{ name为a的需要验证的节点
wrong:"请输入2-5位数字", wrong,right,empty,focus这几个字段为提示内容
right:"",
empty:'输入项不能为空',
focus:"请你输入该项",
required:true, 是否为必填
length:'2~5', 长度：x~y或x~
limit:'number' 插件自带验证函数
}
}
});

自定义验证

HTML代码

和上面的结构一样

JS代码：

validate.addMethod.isDate=function(options){//日期验证
var $element=options.element;
var value=$element.val();
if(!value){
options.tipPlacement($element,options.empty);
return false;
}else if(!isNaN(Date.parse(value))){
options.tipPlacement($element,options.right);
return true;
}else{
options.tipPlacement($element,options.wrong);
return false;
}
};
validate({ 调用验证
id:"nc", 验证器名称,方便以后调用
tipPlacement:function(element,tip){ 提示信息放置函数
element.closest('.common').find('.tip').html(tip);将提示信息放在哪里，element为input节点，tip为提示内容
},
rules:{
'f':{
wrong:"您没有选择",
right:"",
empty:'您没有选择',
focus:"请选择",
required:true,
limit: 'isDate' 自定义方法
}
}
});
